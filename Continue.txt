  -- 1. Créer une vue matérialisée pour les destinations populaires
  CREATE MATERIALIZED VIEW mv_popular_destinations AS
  SELECT
    d.*,
    -- Calcul du score de popularité
    (
      (d.tour_count * 0.25) +
      (d.avg_rating * 5 * 0.20) +
      (d.total_bookings * 0.30) +
      (d.wishlist_count * 0.10) +
      (CASE WHEN d.is_featured THEN 10 ELSE 0 END) +
      (CASE WHEN d.is_trending THEN 5 ELSE 0 END)
    ) as popularity_score,
    -- Saison actuelle
    (
      SELECT ds.season_name
      FROM destination_seasons ds
      WHERE ds.destination_id = d.id
        AND EXTRACT(MONTH FROM CURRENT_DATE) BETWEEN ds.month_start AND
  ds.month_end
      LIMIT 1
    ) as current_season,
    -- Prochains festivals (3 mois)
    (
      SELECT jsonb_agg(festival)
      FROM jsonb_array_elements(d.festivals_events) festival
      WHERE (festival->>'date')::date BETWEEN CURRENT_DATE AND CURRENT_DATE
  + INTERVAL '3 months'
    ) as upcoming_festivals
  FROM destinations d
  WHERE d.is_active = true;

  -- Index sur la vue matérialisée
  CREATE INDEX idx_mv_popular_dest_score ON
  mv_popular_destinations(popularity_score DESC);
  CREATE INDEX idx_mv_popular_dest_featured ON
  mv_popular_destinations(is_featured, popularity_score DESC);

  -- Fonction de rafraîchissement (à appeler périodiquement)
  CREATE OR REPLACE FUNCTION refresh_popular_destinations()
  RETURNS void AS $$
  BEGIN
    REFRESH MATERIALIZED VIEW CONCURRENTLY mv_popular_destinations;
  END;
  $$ LANGUAGE plpgsql;

  -- 2. Créer une fonction pour calculer le score
  CREATE OR REPLACE FUNCTION calculate_destination_popularity_score(dest_id
  INTEGER)
  RETURNS NUMERIC AS $$
  DECLARE
    score NUMERIC;
  BEGIN
    SELECT
      (
        (d.tour_count * 0.25) +
        (d.avg_rating * 5 * 0.20) +
        (d.total_bookings * 0.30) +
        (d.wishlist_count * 0.10) +
        (CASE WHEN d.is_featured THEN 10 ELSE 0 END) +
        (CASE WHEN d.is_trending THEN 5 ELSE 0 END)
      )
    INTO score
    FROM destinations d
    WHERE d.id = dest_id;

    RETURN COALESCE(score, 0);
  END;
  $$ LANGUAGE plpgsql;

  -- 3. Trigger pour mettre à jour automatiquement
  CREATE OR REPLACE FUNCTION update_destination_popularity()
  RETURNS TRIGGER AS $$
  BEGIN
    -- Rafraîchir la vue matérialisée de manière asynchrone
    PERFORM pg_notify('refresh_destinations', NEW.id::text);
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  CREATE TRIGGER trg_destination_stats_updated
  AFTER UPDATE OF tour_count, avg_rating, total_bookings, wishlist_count
  ON destinations
  FOR EACH ROW
  EXECUTE FUNCTION update_destination_popularity();

  import { useState, useEffect } from 'react';
  import { useAuth } from '../context/AuthContext';
  import { destinationService } from '../services/destinationService';

  export function useDestinationLikes() {
    const { user } = useAuth();
    const [likedDestinations, setLikedDestinations] = useState(new Set());

    useEffect(() => {
      loadLikes();
    }, [user]);

    const loadLikes = async () => {
      if (user) {
        // Utilisateur connecté : charger depuis le serveur
        const data = await destinationService.getUserLikedDestinations();
        setLikedDestinations(new Set(data.map(d => d.id)));
      } else {
        // Non connecté : charger depuis localStorage
        const local = JSON.parse(localStorage.getItem('likedDestinations')
  || '[]');
        setLikedDestinations(new Set(local));
      }
    };

    const toggleLike = async (destinationId) => {
      if (user) {
        // Connecté : appel serveur
        await destinationService.toggleLike(destinationId);
      } else {
        // Non connecté : localStorage
        const newLikes = new Set(likedDestinations);
        if (newLikes.has(destinationId)) {
          newLikes.delete(destinationId);
        } else {
          newLikes.add(destinationId);
        }
        localStorage.setItem('likedDestinations',
  JSON.stringify([...newLikes]));
        setLikedDestinations(newLikes);
      }

      loadLikes(); // Recharger
    };

    const isLiked = (destinationId) => likedDestinations.has(destinationId);

    return { likedDestinations, toggleLike, isLiked };
  }

  exports.advancedSearch = async (req, res) => {
    try {
      const {
        query,           // Recherche textuelle
        regions,         // Array de régions
        budgetCategories,
        adventureLevels,
        flags,           // Array: ['unesco', 'wildlife', 'heritage',
  'eco_friendly']
        season,          // 'current' ou mois spécifique
        minRating,
        maxPrice,
        activities       // Array d'activités recherchées
      } = req.body;

      let sqlQuery = `
        SELECT d.*,
          calculate_destination_popularity_score(d.id) as popularity_score
        FROM destinations d
        WHERE d.is_active = true
      `;

      const conditions = [];
      const params = [];
      let paramIndex = 1;

      // Recherche textuelle (full-text search)
      if (query) {
        conditions.push(`(
          to_tsvector('english', d.name || ' ' || COALESCE(d.description,
  ''))
          @@ plainto_tsquery('english', $${paramIndex})
        )`);
        params.push(query);
        paramIndex++;
      }

      // Filtres par région
      if (regions && regions.length > 0) {
        conditions.push(`d.region = ANY($${paramIndex})`);
        params.push(regions);
        paramIndex++;
      }

      // Filtres par budget
      if (budgetCategories && budgetCategories.length > 0) {
        conditions.push(`d.budget_category = ANY($${paramIndex})`);
        params.push(budgetCategories);
        paramIndex++;
      }

      // Filtres par niveau d'aventure
      if (adventureLevels && adventureLevels.length > 0) {
        conditions.push(`d.adventure_level = ANY($${paramIndex})`);
        params.push(adventureLevels);
        paramIndex++;
      }

      // Flags spéciaux
      if (flags && flags.length > 0) {
        const flagConditions = [];
        if (flags.includes('unesco')) flagConditions.push('d.unesco_site =
  true');
        if (flags.includes('wildlife'))
  flagConditions.push('d.wildlife_sanctuary = true');
        if (flags.includes('heritage')) flagConditions.push('d.heritage_site
   = true');
        if (flags.includes('eco_friendly'))
  flagConditions.push('d.eco_friendly = true');
        if (flags.includes('family_friendly'))
  flagConditions.push('d.family_friendly = true');

        if (flagConditions.length > 0) {
          conditions.push(`(${flagConditions.join(' OR ')})`);
        }
      }

      // Note minimale
      if (minRating) {
        conditions.push(`d.avg_rating >= $${paramIndex}`);
        params.push(minRating);
        paramIndex++;
      }

      // Prix maximum
      if (maxPrice) {
        conditions.push(`d.price_range_min <= $${paramIndex}`);
        params.push(maxPrice);
        paramIndex++;
      }

      // Activités recherchées
      if (activities && activities.length > 0) {
        conditions.push(`d.activities && $${paramIndex}`);
        params.push(activities);
        paramIndex++;
      }

      // Saison actuelle
      if (season === 'current') {
        conditions.push(`
          EXISTS (
            SELECT 1 FROM destination_seasons ds
            WHERE ds.destination_id = d.id
              AND EXTRACT(MONTH FROM CURRENT_DATE) BETWEEN ds.month_start
  AND ds.month_end
          )
        `);
      }

      // Construire la requête finale
      if (conditions.length > 0) {
        sqlQuery += ' AND ' + conditions.join(' AND ');
      }

      sqlQuery += ' ORDER BY popularity_score DESC, d.avg_rating DESC LIMIT
  50';

      const result = await db.query(sqlQuery, params);

      res.status(200).json({
        status: 200,
        data: result.rows,
        count: result.rows.length
      });

  4.4 Système de Recommandations

  Étape 4.4.1 : Backend - Algorithme de Recommandations

  Fichier : backend/src/services/recommendationService.js (nouveau)

  // Recommandations basées sur l'historique utilisateur
  async function getPersonalizedRecommendations(userId, limit = 6) {
    // 1. Analyser l'historique de bookings de l'utilisateur
    const userHistory = await db.query(`
      SELECT DISTINCT d.id, d.budget_category, d.adventure_level, d.region,
  d.activities
      FROM bookings b
      JOIN tours t ON b.tour_id = t.id
      JOIN tour_destinations td ON t.id = td.tour_id
      JOIN destinations d ON td.destination_id = d.id
      WHERE b.user_id = $1 AND b.status NOT IN ('cancelled', 'failed')
    `, [userId]);

    if (userHistory.rows.length === 0) {
      // Pas d'historique → retourner les destinations populaires
      return destinationService.getTopDestinations(limit);
    }

    // 2. Extraire les préférences de l'utilisateur
    const preferences = analyzeUserPreferences(userHistory.rows);

    // 3. Trouver des destinations similaires non visitées
    const recommendations = await db.query(`
      SELECT d.*,
        calculate_destination_popularity_score(d.id) as score,
        (
          -- Score de similarité avec les préférences
          (CASE WHEN d.budget_category = $1 THEN 10 ELSE 0 END) +
          (CASE WHEN d.adventure_level = $2 THEN 10 ELSE 0 END) +
          (CASE WHEN d.activities && $3 THEN 15 ELSE 0 END) +
          (CASE WHEN d.region = ANY($4) THEN 5 ELSE 0 END)
        ) as similarity_score
      FROM destinations d
      WHERE d.is_active = true
        AND d.id NOT IN (
          SELECT DISTINCT td.destination_id
          FROM bookings b
          JOIN tours t ON b.tour_id = t.id
          JOIN tour_destinations td ON t.id = td.tour_id
          WHERE b.user_id = $5
        )
      ORDER BY similarity_score DESC, score DESC
      LIMIT $6
    `, [
      preferences.budgetCategory,
      preferences.adventureLevel,
      preferences.activities,
      preferences.regions,
      userId,
      limit
    ]);

    return recommendations.rows;
  }

  function analyzeUserPreferences(history) {
    // Logique pour extraire les préférences dominantes
    const budgetCounts = {};
    const adventureCounts = {};
    const allActivities = new Set();
    const allRegions = new Set();

    history.forEach(dest => {
      budgetCounts[dest.budget_category] =
  (budgetCounts[dest.budget_category] || 0) + 1;
      adventureCounts[dest.adventure_level] =
  (adventureCounts[dest.adventure_level] || 0) + 1;
      dest.activities?.forEach(act => allActivities.add(act));
      if (dest.region) allRegions.add(dest.region);
    });

    return {
      budgetCategory: Object.keys(budgetCounts).reduce((a, b) =>
  budgetCounts[a] > budgetCounts[b] ? a : b),
      adventureLevel: Object.keys(adventureCounts).reduce((a, b) =>
  adventureCounts[a] > adventureCounts[b] ? a : b),
      activities: Array.from(allActivities),
      regions: Array.from(allRegions)
    };
  }

  const redis = require('redis');
  const client = redis.createClient();

  exports.cacheDestinations = (duration = 3600) => {
    return async (req, res, next) => {
      const key = `destinations:${req.originalUrl}`;

      try {
        const cachedData = await client.get(key);
        if (cachedData) {
          return res.json(JSON.parse(cachedData));
        }

        // Override res.json pour cacher la réponse
        const originalJson = res.json.bind(res);
        res.json = (data) => {
          client.setex(key, duration, JSON.stringify(data));
          return originalJson(data);
        };

        next();
      } catch (error) {
        next();
      }
    };
  };
